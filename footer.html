/* ========================================================================== EVENT ENERGY CORE Núcleo de partículas 3D — identidade visual assinatura do site. Cresce, ganha intensidade e gera ondas conforme a navegação/scroll. Perf: menos partículas + menos poligonos em mobile (ver isMobile). ========================================================================== */ (function () { const canvas = document.getElementById('energyCoreCanvas'); if (!canvas || typeof THREE === 'undefined') return; const isMobile = window.matchMedia('(max-width: 768px)').matches; const prefersReducedMotion = window.matchMedia('(prefers-reduced-motion: reduce)').matches; const PARTICLE_COUNT = isMobile ? 900 : 3200; const CORE_RADIUS = 1.6; const scene = new THREE.Scene(); const camera = new THREE.PerspectiveCamera(50, canvas.clientWidth / canvas.clientHeight, 0.1, 100); camera.position.set(0, 0, isMobile ? 6.5 : 5.6); const renderer = new THREE.WebGLRenderer({ canvas, antialias: true, alpha: true }); renderer.setPixelRatio(Math.min(window.devicePixelRatio, isMobile ? 1.5 : 2)); renderer.setSize(canvas.clientWidth, canvas.clientHeight); // ---- Núcleo: nuvem de partículas distribuída numa esfera com jitter ---- const positions = new Float32Array(PARTICLE_COUNT * 3); const basePositions = new Float32Array(PARTICLE_COUNT * 3); const speeds = new Float32Array(PARTICLE_COUNT); const colorMix = new Float32Array(PARTICLE_COUNT); const orange = new THREE.Color('#ff5a1f'); const blue = new THREE.Color('#00b4ff'); const white = new THREE.Color('#f7f7f8'); for (let i = 0; i < PARTICLE_COUNT; i++) { const theta = Math.random() * Math.PI * 2; const phi = Math.acos(2 * Math.random() - 1); const r = CORE_RADIUS * (0.6 + Math.random() * 0.5); const x = r * Math.sin(phi) * Math.cos(theta); const y = r * Math.sin(phi) * Math.sin(theta); const z = r * Math.cos(phi); positions.set([x, y, z], i * 3); basePositions.set([x, y, z], i * 3); speeds[i] = 0.4 + Math.random() * 0.8; colorMix[i] = Math.random(); } const geometry = new THREE.BufferGeometry(); geometry.setAttribute('position', new THREE.BufferAttribute(positions, 3)); const colors = new Float32Array(PARTICLE_COUNT * 3); for (let i = 0; i < PARTICLE_COUNT; i++) { const c = colorMix[i] < 0.72 ? orange : (colorMix[i] < 0.9 ? blue : white); colors.set([c.r, c.g, c.b], i * 3); } geometry.setAttribute('color', new THREE.BufferAttribute(colors, 3)); const material = new THREE.PointsMaterial({ size: isMobile ? 0.028 : 0.022, vertexColors: true, transparent: true, opacity: 0.85, depthWrite: false, blending: THREE.AdditiveBlending, }); const points = new THREE.Points(geometry, material); scene.add(points); // ---- Linhas dinâmicas — conectam partículas próximas ao centro (energia) ---- const lineGeometry = new THREE.BufferGeometry(); const LINE_COUNT = isMobile ? 40 : 140; const linePositions = new Float32Array(LINE_COUNT * 2 * 3); lineGeometry.setAttribute('position', new THREE.BufferAttribute(linePositions, 3)); const lineMaterial = new THREE.LineBasicMaterial({ color: '#ff5a1f', transparent: true, opacity: 0.18 }); const lines = new THREE.LineSegments(lineGeometry, lineMaterial); scene.add(lines); // Luz volumétrica simulada (glow sprite aditivo atrás do núcleo) const glowTexture = (() => { const size = 256; const c = document.createElement('canvas'); c.width = c.height = size; const ctx = c.getContext('2d'); const g = ctx.createRadialGradient(size / 2, size / 2, 0, size / 2, size / 2, size / 2); g.addColorStop(0, 'rgba(255,90,31,0.55)'); g.addColorStop(1, 'rgba(255,90,31,0)'); ctx.fillStyle = g; ctx.fillRect(0, 0, size, size); return new THREE.CanvasTexture(c); })(); const glowMaterial = new THREE.SpriteMaterial({ map: glowTexture, transparent: true, depthWrite: false, blending: THREE.AdditiveBlending }); const glow = new THREE.Sprite(glowMaterial); glow.scale.set(6, 6, 1); scene.add(glow); // ---- Estado de intensidade — controlado pelo scroll (main.js chama isto) ---- let targetIntensity = 0.4; let currentIntensity = 0.4; let targetScale = 1; let currentScale = 1; let coreActive = true; window.EnergyCore = { setIntensity(v) { targetIntensity = THREE.MathUtils.clamp(v, 0, 1); }, setScale(v) { targetScale = THREE.MathUtils.clamp(v, 0.6, 1.8); }, setActive(isActive) { coreActive = isActive; }, }; // ---- Cursor — interação subtil (desktop apenas) ---- const pointer = { x: 0, y: 0 }; if (!isMobile) { window.addEventListener('pointermove', (e) => { pointer.x = (e.clientX / window.innerWidth) * 2 - 1; pointer.y = -(e.clientY / window.innerHeight) * 2 + 1; }); } const clock = new THREE.Clock(); function animate() { requestAnimationFrame(animate); if (!coreActive) return; // poupa CPU/GPU quando não está na Home const t = clock.getElapsedTime(); currentIntensity += (targetIntensity - currentIntensity) * 0.05; currentScale += (targetScale - currentScale) * 0.04; // Rotação cinematográfica lenta da câmara/núcleo points.rotation.y = t * 0.06; points.rotation.x = Math.sin(t * 0.05) * 0.08; lines.rotation.copy(points.rotation); points.scale.setScalar(currentScale); lines.scale.setScalar(currentScale); glow.material.opacity = 0.35 + currentIntensity * 0.4; // Partículas "respiram" para fora do núcleo com a intensidade const posAttr = geometry.attributes.position; for (let i = 0; i < PARTICLE_COUNT; i++) { const i3 = i * 3; const bx = basePositions[i3], by = basePositions[i3 + 1], bz = basePositions[i3 + 2]; const wobble = 1 + Math.sin(t * speeds[i] + i) * 0.06 * (0.5 + currentIntensity); const push = 1 + currentIntensity * 0.35; posAttr.array[i3] = bx * wobble * push; posAttr.array[i3 + 1] = by * wobble * push; posAttr.array[i3 + 2] = bz * wobble * push; } posAttr.needsUpdate = true; // Linhas de energia — pares aleatórios reamostrados lentamente if (Math.floor(t * 4) % 6 === 0) { const lp = lineGeometry.attributes.position.array; for (let i = 0; i < LINE_COUNT; i++) { const a = Math.floor(Math.random() * PARTICLE_COUNT) * 3; lp.set([0, 0, 0], i * 6); lp.set([basePositions[a] * currentScale, basePositions[a + 1] * currentScale, basePositions[a + 2] * currentScale], i * 6 + 3); } lineGeometry.attributes.position.needsUpdate = true; } if (!isMobile) { camera.position.x += (pointer.x * 0.6 - camera.position.x) * 0.03; camera.position.y += (pointer.y * 0.4 - camera.position.y) * 0.03; camera.lookAt(0, 0, 0); } renderer.render(scene, camera); } if (!prefersReducedMotion) { animate(); } else { // Reduced motion: um único frame estático, sem loop contínuo renderer.render(scene, camera); } window.addEventListener('resize', () => { camera.aspect = canvas.clientWidth / canvas.clientHeight; camera.updateProjectionMatrix(); renderer.setSize(canvas.clientWidth, canvas.clientHeight); }); })();
